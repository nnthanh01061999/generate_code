<div className="subheading">Example Usage</div>
<p>
    <code>
        {`import { yupResolver } from '@hookform/resolvers/yup';
import { Button, Form } from 'antd';
import { FormProvider, useForm } from 'react-hook-form';
import * as yup from 'yup';
import InputControl from './input/InputControl';
import InputNumberControl from './input/InputNumberControl';
\ninterface ExampleFormValues {
    name: string;
    age: number;
}
\nfunction Example() {
    const schema = yup.object().shape({
        name: yup.string().required(),
        age: yup.number().required(),
    });\n
    const methods = useForm<ExampleFormValues>({
        defaultValues: { name: '', age: 0 },
        resolver: yupResolver(schema),
    });\n
    const { handleSubmit } = methods;\n
    const onSucess = (data: ExampleFormValues) => {
        console.log(data);
    };\n
    return (
        <FormProvider {...methods}>
            <Form layout="vertical" onFinish={handleSubmit(onSucess)}>
                <InputControl name="name" label="Name" wrapperProps={{ required: true }} />
                <InputNumberControl name="age" label="Age" wrapperProps={{ required: true }} />
                <Button htmlType="submit">Submit</Button>
            </Form>
        </FormProvider>
    );
}\n
export default Example;
`}

</code>

</p>
